middleware.go (Day 4 — logging and the timeout wrapper):

package main

import (
    "context"
    "log/slog"
    "net/http"
    "time"
)

// loggingMiddleware is a decorator: same shape as Day 1's LoggingStore, one
// layer wide, delegates everything else through.
func loggingMiddleware(log *slog.Logger, next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        next.ServeHTTP(w, r)
        log.Info("request", "method", r.Method, "path", r.URL.Path, "took", time.Since(start))
    })
}

// timeoutMiddleware bounds every request's context to d, so nothing downstream
// can block past it — Day 4 section 1's pattern, wired at the handler boundary.
func timeoutMiddleware(d time.Duration, next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        ctx, cancel := context.WithTimeout(r.Context(), d)
        defer cancel()
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}

server.go (Day 4 — handlers, hits now correctly guarded):

package main

import (
    "crypto/rand"
    "encoding/base64"
    "encoding/json"
    "errors"
    "fmt"
    "net/http"
    "sync"
    "time"

    "linkshort/link"
)

type server struct {
    store link.Store
    mu    sync.Mutex
    hits  map[string]int
}

func newServer(store link.Store) *server {
    return &server{store: store, hits: make(map[string]int)}
}

func newCode() (string, error) {
    buf := make([]byte, 4)
    if _, err := rand.Read(buf); err != nil {
        return "", err
    }
    return base64.RawURLEncoding.EncodeToString(buf), nil
}

type createRequest struct {
    URL string `json:"url"`
}

type createResponse struct {
    Code string `json:"code"`
    URL  string `json:"url"`
}

// handleCreate: POST /links {"url": "..."} -> {"code": "...", "url": "..."}
func (s *server) handleCreate(w http.ResponseWriter, r *http.Request) {
    var req createRequest
    if err := json.NewDecoder(r.Body).Decode(&req); err != nil || req.URL == "" {
        http.Error(w, `body must be {"url": "..."}`, http.StatusBadRequest)
        return
    }
    code, err := newCode()
    if err != nil {
        http.Error(w, "internal error", http.StatusInternalServerError)
        return
    }
    l := link.Link{Code: code, URL: req.URL, CreatedAt: time.Now()}
    if err := s.store.Save(r.Context(), l); err != nil {
        http.Error(w, "internal error", http.StatusInternalServerError)
        return
    }
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(http.StatusCreated)
    json.NewEncoder(w).Encode(createResponse{Code: l.Code, URL: l.URL})
}

// resolve wraps the store lookup with request-scoped context — Day 2's
// pattern, carrying whatever deadline the timeout middleware set.
func (s *server) resolve(r *http.Request, code string) (link.Link, error) {
    l, err := s.store.Get(r.Context(), code)
    if err != nil {
        return link.Link{}, fmt.Errorf("resolve %q: %w", code, err)
    }
    return l, nil
}

// handleRedirect: GET /{code} -> 302 to the target URL, counts the hit.
func (s *server) handleRedirect(w http.ResponseWriter, r *http.Request) {
    code := r.PathValue("code")
    l, err := s.resolve(r, code)
    switch {
    case errors.Is(err, link.ErrNotFound):
        http.NotFound(w, r)
        return
    case err != nil:
        http.Error(w, "internal error", http.StatusInternalServerError)
        return
    }

    s.mu.Lock()
    s.hits[code]++
    s.mu.Unlock()

    http.Redirect(w, r, l.URL, http.StatusFound)
}

type statsResponse struct {
    Code string `json:"code"`
    URL  string `json:"url"`
    Hits int    `json:"hits"`
}

// handleStats: GET /{code}/stats -> {"code", "url", "hits"}
func (s *server) handleStats(w http.ResponseWriter, r *http.Request) {
    code := r.PathValue("code")
    l, err := s.resolve(r, code)
    switch {
    case errors.Is(err, link.ErrNotFound):
        http.NotFound(w, r)
        return
    case err != nil:
        http.Error(w, "internal error", http.StatusInternalServerError)
        return
    }
    s.mu.Lock()
    hits := s.hits[code]
    s.mu.Unlock()
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(statsResponse{Code: l.Code, URL: l.URL, Hits: hits})
}

link/postgres.go (Day 5–6 — unchanged from yesterday, given again for completeness):

package link

import (
    "context"
    "errors"

    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgxpool"
)

var _ Store = (*PostgresStore)(nil)

type PostgresStore struct {
    pool *pgxpool.Pool
}

func NewPostgresStore(ctx context.Context, connString string) (*PostgresStore, error) {
    pool, err := pgxpool.New(ctx, connString)
    if err != nil {
        return nil, err
    }
    return &PostgresStore{pool: pool}, nil
}

func (s *PostgresStore) Save(ctx context.Context, l Link) error {
    _, err := s.pool.Exec(ctx,
        `INSERT INTO links (code, url) VALUES ($1, $2)
         ON CONFLICT (code) DO UPDATE SET url = EXCLUDED.url`,
        l.Code, l.URL,
    )
    return err
}

func (s *PostgresStore) Get(ctx context.Context, code string) (Link, error) {
    var l Link
    err := s.pool.QueryRow(ctx,
        `SELECT code, url, created_at FROM links WHERE code = $1`, code,
    ).Scan(&l.Code, &l.URL, &l.CreatedAt)
    if errors.Is(err, pgx.ErrNoRows) {
        return Link{}, ErrNotFound
    }
    if err != nil {
        return Link{}, err
    }
    return l, nil
}

main.go (new today — the first file where Day 4's server and Day 5–6's Postgres store actually meet):

package main

import (
    "context"
    "errors"
    "log/slog"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"

    "linkshort/link"
)

func main() {
    ctx := context.Background()

    var store link.Store
    if dsn := os.Getenv("LINKSHORT_DB_URL"); dsn != "" {
        pg, err := link.NewPostgresStore(ctx, dsn)
        if err != nil {
            panic(err)
        }
        store = pg
    } else {
        store = link.NewMemStore() // no LINKSHORT_DB_URL set — fine for today's tests
    }
    srv := newServer(link.NewLoggingStore(store, slog.Default()))

    mux := http.NewServeMux()
    mux.HandleFunc("POST /links", srv.handleCreate)
    mux.HandleFunc("GET /{code}", srv.handleRedirect)
    mux.HandleFunc("GET /{code}/stats", srv.handleStats)

    var handler http.Handler = mux
    handler = timeoutMiddleware(2*time.Second, handler)
    handler = loggingMiddleware(slog.Default(), handler)

    httpServer := &http.Server{Addr: ":8080", Handler: handler}

    shutdownCtx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
    defer stop()

    go func() {
        if err := httpServer.ListenAndServe(); err != nil && !errors.Is(err, http.ErrServerClosed) {
            slog.Error("server error", "err", err)
        }
    }()

    <-shutdownCtx.Done()
    stopCtx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()
    httpServer.Shutdown(stopCtx)
}

1 · Table-driven tests & subtests

A table-driven test is one function, one loop, many cases — instead of copy-pasting TestX, TestY, TestZ with the same three lines each. t.Run(name, func(t *testing.T) {...}) spins up a subtest per case: each gets its own pass/fail line and its own name in the output, and go test -run TestMemStoreTable/empty_url_segment can target exactly one case.

Add this to the existing link/store_test.go — same package, same imports already in that file, nothing new to add:

func TestMemStoreTable(t *testing.T) {
    cases := []struct {
        name string
        code string
        url  string
    }{
        {"short code", "go", "https://golang.org"},
        {"url with a path", "ex", "https://example.com/docs"},
    }
    for _, tc := range cases {
        t.Run(tc.name, func(t *testing.T) {
            s := NewMemStore()
            if err := s.Save(context.Background(), Link{Code: tc.code, URL: tc.url}); err != nil {
                t.Fatal(err)
            }
            got, err := s.Get(context.Background(), tc.code)
            if err != nil {
                t.Fatal(err)
            }
            if got.URL != tc.url {
                t.Fatalf("got %q, want %q", got.URL, tc.url)
            }
        })
    }
}

Notice each subtest builds its own fresh NewMemStore() — cases share the loop, not state. A case that mutated a shared store would make later cases depend on earlier ones passing first, which defeats the point of naming them independently.

2 · httptest: testing handlers without a port

net/http/httptest lets a test call a handler directly — no ListenAndServe, no real socket, no port to clash with another test running in parallel. httptest.NewRequest builds an *http.Request in memory; httptest.NewRecorder is an http.ResponseWriter that captures the status code, headers, and body instead of writing them to a real connection. Route through the actual mux, not the bare handler function — r.PathValue("code") is only populated once ServeMux has matched the route:

package main

import (
    "context"
    "net/http"
    "net/http/httptest"
    "testing"

    "linkshort/link"
)

func TestHandleRedirect(t *testing.T) {
    ctx := context.Background()
    store := link.NewMemStore()
    if err := store.Save(ctx, link.Link{Code: "go", URL: "https://golang.org"}); err != nil {
        t.Fatal(err)
    }
    srv := newServer(store)
    mux := http.NewServeMux()
    mux.HandleFunc("GET /{code}", srv.handleRedirect)

    req := httptest.NewRequest(http.MethodGet, "/go", nil)
    rec := httptest.NewRecorder()
    mux.ServeHTTP(rec, req)

    if rec.Code != http.StatusFound {
        t.Fatalf("status = %d, want %d", rec.Code, http.StatusFound)
    }
    if got := rec.Header().Get("Location"); got != "https://golang.org" {
        t.Fatalf("Location = %q, want %q", got, "https://golang.org")
    }
}

3 · Integration tests against real Postgres

An integration test for PostgresStore can't run against MemStore's fast in-memory world — it needs a real database. Without Docker, the substitute for a disposable testcontainer is a Neon branch: an instant, free copy of your linkshort database, safe to write throwaway rows into and delete. A build tag keeps this test out of your everyday, fast go test ./...:

//go:build integration

package main

import (
    "context"
    "os"
    "testing"
    "time"

    "linkshort/link"
)

func TestPostgresStoreSaveThenGet(t *testing.T) {
    dsn := os.Getenv("LINKSHORT_TEST_DB_URL")
    if dsn == "" {
        t.Skip("LINKSHORT_TEST_DB_URL not set — point it at a throwaway Neon branch")
    }
    ctx := context.Background()
    store, err := link.NewPostgresStore(ctx, dsn)
    if err != nil {
        t.Fatal(err)
    }
    want := link.Link{Code: "it-test", URL: "https://go.dev", CreatedAt: time.Now()}
    if err := store.Save(ctx, want); err != nil {
        t.Fatal(err)
    }
    got, err := store.Get(ctx, want.Code)
    if err != nil {
        t.Fatal(err)
    }
    if got.Code != want.Code || got.URL != want.URL {
        t.Fatalf("got %+v, want %+v", got, want)
    }
}

Save this as postgres_integration_test.go. The //go:build integration line must be the first line of the file, followed by a blank line before package main — Go's build-tag parser requires that exact shape. t.Skip rather than t.Fatal when the env var is unset means the test suite still passes green in CI or on a laptop with no Neon branch configured — skipped, not failed, is the honest signal for "this test needs infrastructure I don't have right now."

Neon branch, five minutes: in the Neon console, open your linkshort project → Branches → New branch from main (this copies today's schema and data instantly, copy-on-write — no dump/restore). Copy its connection string into LINKSHORT_TEST_DB_URL. Run go test -tags=integration ./...; delete the branch afterward, or keep it and re-run against it tomorrow — either way, main stayed untouched.

4 · Benchmarks: measuring, not guessing

A benchmark answers "how fast," a test answers "how correct" — different question, different function shape, same file is fine. Go 1.24 added b.Loop(), now the preferred shape over the classic for i := 0; i < b.N; i++: it fixes a longstanding footgun where the compiler could sometimes optimize away work inside the old loop entirely (nothing appeared to use the result), silently reporting a bogus near-zero time.

package main

import (
    "context"
    "testing"

    "linkshort/link"
)

func BenchmarkMemStoreGet(b *testing.B) {
    s := link.NewMemStore()
    s.Save(context.Background(), link.Link{Code: "go", URL: "https://golang.org"})

    for b.Loop() {
        s.Get(context.Background(), "go")
    }
}

Save this as bench_test.go. Setup (the Save above) runs once, outside the loop — only the code inside for b.Loop() is timed, and the testing package runs it enough iterations to get a stable measurement, adjusting automatically the way b.N used to require you to reason about by hand. Run it with:

go test -bench=. -benchmem ./...

-benchmem adds allocations-per-op alongside time-per-op — for MemStore.Get, expect something close to zero allocations; a map read shouldn't need to allocate anything.

5 · Tooling: vet, fix, and a taste of pprof

go vet ./... catches real bugs that still compile — a Printf verb that doesn't match its argument, a struct copied by value that contains a mutex, a range variable captured wrong. Run it before every commit; it's fast and it's been silent through this whole week for a reason.

go fix is Go's built-in modernizer: it rewrites code to newer, equivalent idioms it's certain about — a hand-rolled min/max helper collapsed into the 1.21 builtin, an old three-clause counting loop turned into for i := range n (1.22). Run go fix ./... against the whole linkshort tree; it's a no-op if there's nothing to modernize, and it never changes behavior — only style it's certain is equivalent.

pprof is where "it feels slow" turns into a number. One line added to a benchmark run produces a CPU profile:

go test -bench=. -cpuprofile=cpu.out ./...
go tool pprof cpu.out
(pprof) top10

top10 lists the functions eating the most CPU time in that run. For today's tiny MemStore.Get benchmark this will look almost empty — that's fine, the point today is knowing the three-command shape exists and reaching for it the moment a benchmark number looks worse than expected, not memorizing flame-graph reading in one sitting.
