link/memstore.go (replace the whole file — Day 8's mutex-guarded store):

package link

import (
    "context"
    "sync"
)

var _ Store = (*MemStore)(nil)

type MemStore struct {
    mu    sync.RWMutex
    links map[string]Link
}

func NewMemStore() *MemStore {
    return &MemStore{links: make(map[string]Link)}
}

func (s *MemStore) Save(ctx context.Context, l Link) error {
    s.mu.Lock()
    defer s.mu.Unlock()
    s.links[l.Code] = l
    return nil
}

func (s *MemStore) Get(ctx context.Context, code string) (Link, error) {
    s.mu.RLock()
    defer s.mu.RUnlock()
    l, ok := s.links[code]
    if !ok {
        return Link{}, ErrNotFound
    }
    return l, nil
}

New file link/counter.go (Day 8's click counter):

package link

import "sync"

// ClickCounter is a concurrency-safe tally of redirects per code.
type ClickCounter struct {
    mu     sync.Mutex
    counts map[string]int
}

func NewClickCounter() *ClickCounter {
    return &ClickCounter{counts: make(map[string]int)}
}

func (c *ClickCounter) Inc(code string) {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.counts[code]++
}

func (c *ClickCounter) Count(code string) int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.counts[code]
}

func (c *ClickCounter) Snapshot() map[string]int {
    c.mu.Lock()
    defer c.mu.Unlock()
    out := make(map[string]int, len(c.counts))
    for k, v := range c.counts {
        out[k] = v
    }
    return out
}

checker.go (replace the whole file — Day 9's worker pool + Day 10's ctx/errgroup rewrite):

package main

import (
    "context"
    "strings"
    "time"

    "linkshort/link"

    "golang.org/x/sync/errgroup"
)

type result struct {
    code string
    ok   bool
}

func checkOne(l link.Link) result {
    time.Sleep(50 * time.Millisecond)
    return result{code: l.Code, ok: strings.HasPrefix(l.URL, "https://")}
}

const maxWorkers = 4

func checkAll(ctx context.Context, links []link.Link) map[string]bool {
    jobs := make(chan link.Link)
    results := make(chan result)

    g, ctx := errgroup.WithContext(ctx)
    for w := 0; w < maxWorkers; w++ {
        g.Go(func() error {
            for {
                select {
                case l, open := <-jobs:
                    if !open {
                        return nil
                    }
                    select {
                    case results <- checkOne(l):
                    case <-ctx.Done():
                        return ctx.Err()
                    }
                case <-ctx.Done():
                    return ctx.Err()
                }
            }
        })
    }

    go func() {
        defer close(jobs)
        for _, l := range links {
            select {
            case jobs <- l:
            case <-ctx.Done():
                return
            }
        }
    }()

    go func() {
        g.Wait()
        close(results)
    }()

    out := make(map[string]bool, len(links))
    for {
        select {
        case r, open := <-results:
            if !open {
                return out
            }
            out[r.code] = r.ok
        case <-ctx.Done():
            return out
        }
    }
}

1 · Goroutine leaks: the bug that doesn't crash

A goroutine leak is the most Go-specific bug there is, and the most dangerous, because nothing crashes. A leaked goroutine just sits there — a few KB of stack, a blocked channel op — invisible in a quick test run, and fatal at scale: ten thousand requests, ten thousand leaked goroutines, memory climbing until the process is killed.

// LEAK: nobody ever reads from ch. This goroutine blocks on the send
// forever — it is never collected, even though its result is never used.
func fetchWithTimeout(ctx context.Context) int {
    ch := make(chan int)
    go func() {
        result := slowCompute()
        ch <- result   // blocks forever if the caller already gave up below
    }()

    select {
    case v := <-ch:
        return v
    case <-ctx.Done():
        return -1   // caller moves on — but the goroutine above is still stuck
    }
}

The tell: an unbuffered channel with exactly one sender and exactly one receiver, where the receiver has a second way out (ctx.Done()) that the sender doesn't share. The fix is Day 10's whole lesson applied one level deeper — give the sender the same escape hatch:

go func() {
    result := slowCompute()
    select {
    case ch <- result:
    case <-ctx.Done():   // FIXED — sender can also give up
    }
}()

How to actually catch this in review or in CI, not just in theory: runtime.NumGoroutine() before and after a suspect operation (a rising baseline across repeated calls is a leak), or the standard library's own go.uber.org/goleak in tests — call defer goleak.VerifyNone(t) at the top of a test and it fails if any goroutine outlives it. linkshort stays stdlib-only, so today's practice below uses the NumGoroutine technique instead of adding a new dependency.

2 · Deadlock: when Go tells you outright

A deadlock is a leak's loud cousin: instead of one goroutine stuck while the rest of the program carries on, every goroutine is stuck. Go's runtime actually notices this specific case — nothing left that could ever become runnable — and kills the process outright:

func main() {
    ch := make(chan int)
    ch <- 1   // unbuffered send, no receiver anywhere — nothing else is running
}
// fatal error: all goroutines are asleep - deadlock!

That crash is the friendly case — loud, immediate, a stack trace pointing at the exact blocked line. The dangerous version is a partial deadlock: two goroutines stuck waiting on each other, while a third goroutine (an HTTP server's accept loop, say) is still very much alive and serving other requests. The runtime sees forward progress happening somewhere, so it does not detect this — the process looks healthy while two goroutines are silently wedged forever.

// classic two-lock deadlock: goroutine A locks mu1 then wants mu2;
// goroutine B locks mu2 then wants mu1. Whichever order runs first,
// the second goroutine to start now waits on a lock the first will
// never release, because the first is waiting right back on it.
go func() { mu1.Lock(); defer mu1.Unlock(); mu2.Lock(); defer mu2.Unlock() /* ... */ }()
go func() { mu2.Lock(); defer mu2.Unlock(); mu1.Lock(); defer mu1.Unlock() /* ... */ }()

The fix is a rule, not a clever trick: always acquire multiple locks in the same, fixed order, everywhere in the codebase — e.g. always mu1 before mu2, never the reverse. Neither goroutine can then be the one holding the second lock the other wants first. It's boring and it's the actual answer interviewers want, not "use a bigger lock."

3 · The classic bug patterns

Four shapes come up over and over — in code review, in interviews, in production incidents. Recognize them on sight.

Send on a closed channel — panics, always.

ch := make(chan int)
close(ch)
ch <- 1   // panic: send on closed channel

The rule from Day 3 was "the sender closes" — today's sharper version: only the sole sender may close, and only after every send is done. The moment more than one goroutine can send on the same channel, none of them can safely be the one to call close — any other sender might still be mid-send. This is exactly why Day 9's worker pool has the feeder close jobs (it's the only sender) and the closer goroutine — not any individual worker — close results (workers are the senders, multiple of them; only after g.Wait() confirms all of them are done does closing become safe).

Closing twice — also panics, always.

close(ch)
close(ch)   // panic: close of closed channel

Comes up most often when a channel gets closed on more than one exit path — an error branch and a success branch both closing "just in case." The fix is the same discipline as above: exactly one place in the code owns closing any given channel, full stop, and every other path either doesn't close it or communicates through a different mechanism (a done chan struct{} that itself only gets closed once, or a sync.Once — Day 12's topic — wrapping the close).

Pre-1.22 loop-variable capture — a correctness bug, not a panic.

// Go < 1.22: ALL goroutines below could print the SAME final value of l,
// because pre-1.22 every iteration reused ONE shared `l` variable.
for _, l := range links {
    go func() {
        fmt.Println(l.Code)   // pre-1.22: probably prints the LAST code, several times over
    }()
}

Go 1.22 (Feb 2024) fixed this at the language level — every loop iteration now gets its own fresh l, and this exact code is correct as written on any Go you're running this week. It's still worth recognizing on sight for two reasons: reading pre-1.22 code (plenty still in production) and interviews, where "what does this print, and why did Go change it in 1.22" is a favorite question. The old-style defensive fix, l := l or passing l as a parameter (exactly what Day 3's go func(l link.Link) { ... }(l) did), still works today — it's just no longer required.

Forgotten wg.Done() — the WaitGroup version of a deadlock.

var wg sync.WaitGroup
for i := 0; i < n; i++ {
    wg.Add(1)
    go func() {
        // forgot: defer wg.Done()
        doWork()
    }()
}
wg.Wait()   // blocks forever — the counter never reaches zero

Every early return or panic inside the goroutine skips a bare wg.Done() written at the bottom of the function — exactly the same "forgot one exit path" failure defer mu.Unlock() prevents for locks. The rule is identical: defer wg.Done() on the line right after the goroutine starts, never a bare call at the end. Day 8 through 10's checker.go has done this correctly every single time — that's not an accident, it's the reflex today names explicitly.

4 · -race as a reflex, not a last resort

Every prior lesson this week ended with go test -race ./... as a final check. Today reframes it: run it constantly while writing concurrent code, not once at the end — the race detector doesn't just confirm a hunch, it finds bugs you weren't even looking for, with the exact two stack traces of the colliding accesses. It costs real overhead (5–10x slower, more memory) so it's not for production traffic, but for local dev and CI on anything touching goroutines, it should run by default, every time, the same reflex as go vet. One more thing worth knowing for interviews: -race only flags races it actually observes during that run — it proves a race happened, never that one can't. A clean -race run is evidence, not a guarantee.

import (
    "context"
    "fmt"
    "runtime"
    "testing"
    "time"

    "linkshort/link"
)

// TestCheckAllCancelledLeavesNoGoroutinesBehind cancels ctx mid-sweep, waits
// briefly for things to settle, then checks the goroutine count returned to
// baseline. With Step 1's bug in place, a worker can be stuck forever on
// `results <- checkOne(l)` after every reader has walked away — the count
// stays elevated instead of returning to baseline.
func TestCheckAllCancelledLeavesNoGoroutinesBehind(t *testing.T) {

    }

    before := runtime.NumGoroutine()

        time.Sleep(5 * time.Millisecond)
        cancel()
