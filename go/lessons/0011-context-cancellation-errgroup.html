package main

import (
    "strings"
    "sync"
    "time"

    "linkshort/link"
)

type result struct {
    code string
    ok   bool
}

func checkOne(l link.Link) result {
    time.Sleep(50 * time.Millisecond)
    return result{code: l.Code, ok: strings.HasPrefix(l.URL, "https://")}
}

const maxWorkers = 4

func checkAll(links []link.Link, timeout time.Duration) map[string]bool {
    jobs := make(chan link.Link)
    results := make(chan result)

    var wg sync.WaitGroup
    for w := 0; w < maxWorkers; w++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for l := range jobs {
                results <- checkOne(l)
            }
        }()
    }

    go func() {
        for _, l := range links {
            jobs <- l
        }
        close(jobs)
    }()

    go func() {
        wg.Wait()
        close(results)
    }()

    out := make(map[string]bool, len(links))
    deadline := time.After(timeout)
    for {
        select {
        case r, open := <-results:
            if !open {
                return out
            }
            out[r.code] = r.ok
        case <-deadline:
            return out
        }
    }
}

1 · What Day 4 promised, and yesterday didn't keep

Day 4 taught context.Context as one cancellation signal, composable across a whole call chain: any caller — an HTTP middleware, a test, another goroutine — sets a deadline, and everything downstream checking ctx.Done() obeys it without knowing who set it. Yesterday's checkAll quietly stepped backward from that: timeout time.Duration is a value only checkAll itself can turn into a deadline, with its own private time.After. A caller can't share an existing deadline with it, can't cancel it early for any other reason ("the HTTP client hung up," "a parent operation failed"), and — Section 5's own admission — can't make the workers actually stop, only make checkAll stop waiting for them.

Today checkAll takes a ctx context.Context instead, the same parameter shape every Store method has had since Day 1.

2 · Threading ctx through a pipeline, not just one call

Day 4's uses of ctx were all a single select guarding one blocking op. A worker pool has several — the feeder's send into jobs, each worker's send into results — and every single one needs its own way out, or cancelling ctx only stops the ones you remembered.

// yesterday — only checkAll's own collecting loop knew about a deadline
func checkAll(links []link.Link, timeout time.Duration) map[string]bool { /* ... */ }

// today — the SAME ctx a caller already has (an HTTP request, a test,
// another cancelled operation) flows all the way to every goroutine
func checkAll(ctx context.Context, links []link.Link) map[string]bool { /* ... */ }

The caller now owns the deadline: ctx, cancel := context.WithTimeout(parent, time.Second) before calling, exactly like Day 4's HTTP handlers. checkAll stops deciding "how long is too long" and starts just obeying whatever ctx it's handed — the composability Day 4 was actually about.

3 · errgroup: workers, cancellation and Wait in one type

Day 4 name-dropped errgroup but never used it — linkshort stayed stdlib-only. Today it earns its spot: a worker pool is exactly "launch N goroutines, wait for all of them, care if any of them fails" — a plain sync.WaitGroup gives you the first two but nothing for the third or for cancellation.

import "golang.org/x/sync/errgroup"

g, ctx := errgroup.WithContext(ctx)   // shadows the parameter on purpose — see below

for w := 0; w < maxWorkers; w++ {
    g.Go(func() error {
        // worker body — returns nil on normal exit, ctx.Err() if cancelled
        return nil
    })
}

if err := g.Wait(); err != nil {
    // first error from any worker (here: usually context.Canceled)
}

errgroup.WithContext(ctx) returns a derived context: this new ctx closes its Done() the instant either the original ctx is cancelled (caller gave up) or any g.Go function returns a non-nil error (one worker failed, cancel the rest) — whichever comes first. None of today's workers can actually fail (checkOne has no error path), so the second trigger never fires in practice; what errgroup earns its keep for today is bundling "N goroutines + a shared derived ctx + a single blocking Wait" into one type instead of a WaitGroup plus hand-rolled cancellation plumbing. Re-assigning ctx to the derived one (g, ctx := errgroup.WithContext(ctx)) is deliberate and idiomatic: every line below should watch the derived signal, not the original.

4 · Closing the leak: a cancellation-aware select on every blocking op

Renaming the parameter isn't the fix by itself. ctx being cancelled does nothing on its own — a chan send or chan receive only reacts to a closed Done() if it's sitting in a select that's actually watching it. Miss one, and that goroutine blocks exactly as before.

// LEAK: if ctx is cancelled while no one reads results anymore,
// this send blocks forever — the goroutine is never collected.
results <- checkOne(l)

// FIXED: a cancellation-aware select — either the send completes,
// or ctx.Done() wins and the goroutine returns instead.
select {
case results <- checkOne(l):
case <-ctx.Done():
    return ctx.Err()
}

That's the whole technique — a cancellation-aware select — applied to every place today's pipeline can block: each worker's receive from jobs, each worker's send into results, and the feeder's send into jobs. Three spots, same one-line pattern each time. Skip any one of them and Day 9's exact gap reopens there.

package main

import (
    "context"
    "strings"
    "time"

    "linkshort/link"

    "golang.org/x/sync/errgroup"
)

type result struct {
    code string
    ok   bool
}

func checkOne(l link.Link) result {
    time.Sleep(50 * time.Millisecond)
    return result{code: l.Code, ok: strings.HasPrefix(l.URL, "https://")}
}

const maxWorkers = 4

func checkAll(ctx context.Context, links []link.Link) map[string]bool {
    jobs := make(chan link.Link)
    results := make(chan result)

    g, ctx := errgroup.WithContext(ctx)
    for w := 0; w < maxWorkers; w++ {
        // TODO 1: g.Go(func() error { ... }) — loop forever, select on:
        //   receive `l, open := <-jobs`: if !open, return nil; else send
        //     checkOne(l) into results — itself a select against ctx.Done()
        //   receive from ctx.Done(): return ctx.Err()
        //   (Section 4 has both selects almost verbatim.)
    }

    go func() {
        // TODO 2: feeder — defer close(jobs), then for _, l := range links,
        //   select { case jobs <- l: / case <-ctx.Done(): return }
    }()

    go func() {              // GIVEN — the closer, same shape as Day 9
        g.Wait()
        close(results)
    }()

    out := make(map[string]bool, len(links))   // GIVEN — collecting loop, ctx.Done() replaces deadline
    for {
        select {
        case r, open := <-results:
            if !open {
                return out
            }
            out[r.code] = r.ok
        case <-ctx.Done():
            return out
        }
    }
}

Step 3 — prove cancellation is real, not just "stops waiting." Add a test where ctx is cancelled while checks are still in flight, and assert checkAll returns close to the cancel time, not close to how long the full sweep would've taken:

func TestCheckAllCancelledContextReturnsPromptly(t *testing.T) {
    links := make([]link.Link, 20)   // 20 links / 4 workers = ~250ms if left to finish
    for i := range links {
        links[i] = link.Link{Code: fmt.Sprintf("l%d", i), URL: "https://x.dev"}
    }
    ctx, cancel := context.WithCancel(context.Background())
    go func() {
        time.Sleep(20 * time.Millisecond)   // cancel well before the sweep would finish
        cancel()
    }()
    start := time.Now()
    got := checkAll(ctx, links)
    elapsed := time.Since(start)
    if elapsed > 100*time.Millisecond {
        t.Fatalf("elapsed %v — checkAll kept running after ctx was cancelled", elapsed)
    }
    if len(got) == 20 {
        t.Fatal("expected a partial map from the cancelled sweep, got all 20")
    }
}
